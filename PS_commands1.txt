#Отправить GET-запрос (cURL)
curl -X GET "https://api.example.com/data"

#Отправить POST-запрос с JSON (cURL)
curl -X POST "https://api.example.com/create" -H "Content-Type: application/json" -d '{"name": "John", "age": 30}'

#Отправить PUT/PATCH-запрос (cURL)
curl -X PUT "https://api.example.com/update/1" -H "Content-Type: application/json" -d '{"name": "John Doe"}'

#Отправить DELETE-запрос (cURL)
curl -X DELETE "https://api.example.com/delete/1"

#Генерация JWT-токена (если есть jq)
echo '{"sub": "123", "name": "John"}' | jq -R -s '{"token": (. | @base64)}'

#Отправить запрос через SOCKS5 (например, Tor)
curl --socks5-hostname localhost:9050 "https://api.example.com"

#Проверить CORS-заголовки API
